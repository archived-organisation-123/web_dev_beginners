HTML: HyperText Markup Language

  • Language that web browsers use to render pages
  • First released in 1993
  • Designed to create web documents. Has since been extended for web applications

Cascading Style Sheets

Functions - Event handlers


    var reds = document.getElementByClassName("red-btn")
    var red = reds.item(0)
    var blues = document.getElementByClassName("blue-btn")
    var blue = blues.item(0)

    // event is populated with information about the click
    function shout(event) {
        var clickedButton = event.target

        alert(clickedButton.classList.toString())
    }

    red.onclick = shout
    blue.onclick = shout

Exercise 4 - On board element click

Modify function onBoardPieceClick to add a class to clicked board piece (available in event.target). The class name should be equal to value of variable turn created in Exercise 1.

Exercise 4 - Solution


    function onBoardPieceClick(event) {
        var boardPiece = event.target
        boardPiece.classList.add(turn)
    }
        
    for (var i = 0; i < boardPieces.length; i = i + 1) {
        boardPieces.item(i).onclick = onBoardPieceClick
    }

If Statements - Make decistions

    if (light == "red") {
